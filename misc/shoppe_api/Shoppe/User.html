    Attempt to authenticate a user based on email & password. Returns the user if successful otherwise returns false.

    @param email_address [String] @param paassword [String] @return [Shoppe::User]

def self.authenticate(email_address, password)
  user = where(email_address: email_address).first
  return false if user.nil?
  return false unless user.authenticate(password)
  user
end
